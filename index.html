  int motorRightPin = 7;
  int motorLeftPin = 6;
  
  int ldrPin = 0;
  int ldrValor = 0;
  
  void setup() {
    pinMode(motorRightPin, OUTPUT);
    pinMode(motorLeftPin, OUTPUT);
    Serial.begin(9600);
  }
  
  void loop() {
    ldrValor = analogRead(ldrPin);
    
    motores(ldrValor);
    
    Serial.println(ldrValor);
    delay(100);
  }
  
  
  void motores(int ldrValor){
    if(ldrValor >= 90){
      digitalWrite(motorLeftPin, HIGH);
      digitalWrite(motorRightPin, LOW);
    } else if(ldrValor <= 80){
      digitalWrite(motorRightPin, HIGH);
      digitalWrite(motorLeftPin, LOW);
    } else {
      digitalWrite(motorRightPin, HIGH);
      digitalWrite(motorLeftPin, HIGH);
    }
  }

Código final:

Definição das variáveis constantes para o sistema, atribuição das portas para as variaveis PIN para o LED, o motor e sensores

Função responsável por definir as configurações do led

Função responsável por dar inicio ao sistema

A função followLineMEF() é a responsável por fazer as verificações do nosso motor, primeiro temos um loop dentro dela onde irá indicar o estado dela e as verificações, são leituras dos sensores comparado com TRESHHOLD que é o indicador da linha branca ou a linha que ele terá que seguir, e também dentro dela é onde ocorre o calculo de erro baseado no PID, caso ela caia em alguma validação ela chama a funcao motorOption()

A motorOption() recebe como parâmetros a opção escolhida, a velocidade do motor esquerdo e do direito e passa para a função motorControl()

Na motorControl() realizamos o controle dos motores definimos os pinos e qual será a função, que no caso OUTPUT é saída e assim definimos qual motor deve ligar e qual deve desligar e qual deve ser a sua velocidade
